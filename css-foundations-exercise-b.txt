** start of styles.css **

/* 1. All odd elements with class 'odd' should have a yellow background */
.odd {
  background-color: yellow;
}

/* 2. The second element (ID 'second') has blue text and a font size of 36px */
#second {
  color: blue;
  font-size: 36px;
}

/* 3. The third element has a font size of 24px, set via class 'third' */
.third {
  font-size: 24px;
}

/* 4. The fourth element (ID 'fourth') has:
   - background color red
   - font size 24px (via class 'fourth')
   - font weight bold (via ID 'fourth')
*/
#fourth {
  font-weight: bold;
  background-color: red;
}

.fourth {
  font-size: 24px;
}

** end of styles.css **
